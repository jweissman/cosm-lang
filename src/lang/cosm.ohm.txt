Cosm {
  Program
    = StatementList

  StatementList
    = Statement StatementTail* ";"?

  StatementTail
    = ";" Statement

  Statement
    = ClassStmt
    | DefStmt
    | LetStmt
    | BareCallStmt
    | Exp

  BareCallStmt
    = BareCallee BareCallArgs

  BareCallee
    = ident BareCalleeTail*

  BareCalleeTail
    = "." name

  ClassStmt
    = "class" ident ClassSuper? "do" ClassBody "end"

  ClassSuper
    = "<" ident

  ClassBody
    = ClassMember*

  ClassMember
    = ClassDefStmt ";"?

  ClassDefStmt
    = "def" "self" "." ident "(" ParamList? ")" "do" StatementList "end"  -- class
    | "def" ident "(" ParamList? ")" "do" StatementList "end"              -- instance

  DefStmt
    = "def" ident "(" ParamList? ")" "do" StatementList "end"

  LetStmt
    = "let" ident "=" Exp

  Exp
    = IfExp

  IfExp
    = "if" Exp "then" StatementList "else" StatementList "end"  -- full
    | OrExp

  OrExp
    = OrExp "||" AndExp   -- or
    | AndExp

  AndExp
    = AndExp "&&" CmpExp  -- and
    | CmpExp

  CmpExp
    = AddExp "==" AddExp  -- eq
    | AddExp "!=" AddExp  -- neq
    | AddExp "<=" AddExp  -- lte
    | AddExp "<" AddExp   -- lt
    | AddExp ">=" AddExp  -- gte
    | AddExp ">" AddExp   -- gt
    | AddExp

  AddExp
    = AddExp "+" MulExp  -- plus
    | AddExp "-" MulExp  -- minus
    | MulExp

  MulExp
    = MulExp "*" ExpExp  -- times
    | MulExp "/" ExpExp  -- divide
    | ExpExp

  ExpExp
    = UnaryExp "^" ExpExp  -- power
    | UnaryExp

  UnaryExp
    = "!" UnaryExp   -- not
    | "+" UnaryExp   -- pos
    | "-" UnaryExp   -- neg
    | PostExp

  PostExp
    = PostExp "." name  -- access
    | PostExp "(" CallArgs? ")"  -- call
    | PriExp

  PriExp
    = "(" Exp ")"  -- paren
    | "do" StatementList "end"  -- block
    | "->" "(" ParamList? ")" "{" Exp "}"  -- lambda
    | "[" ArrayItems? "]"  -- array
    | "{" HashItems? "}"  -- hash
    | string
    | symbol
    | boolean
    | self
    | ivar
    | ident
    | number

  symbol
    = ":" name

  CallArgs
    = Exp ("," Exp)*

  BareCallArgs
    = BareArg ("," BareArg)*

  BareArg
    = string
    | symbol
    | boolean
    | self
    | ivar
    | ident
    | number
    | BareArray
    | BareHash

  BareArray
    = "[" ArrayItems? "]"

  BareHash
    = "{" HashItems? "}"

  ParamList
    = ident ("," ident)*

  ArrayItems
    = Exp ("," Exp)*

  HashItems
    = HashEntry ("," HashEntry)*

  HashEntry
    = ident ":" Exp

  string
    = dstring
    | sstring

  dstring
    = "\"" stringPart* "\""

  sstring
    = "'" sstringPart* "'"

  stringPart
    = interpolation  -- interp
    | stringText     -- text
    | "\\" escape    -- escape

  sstringPart
    = sstringText   -- text
    | "\\" sescape  -- escape

  interpolation
    = "#{" applySyntactic<Exp> "}"

  stringText
    = ~("\\" | "\"" | "#") any  -- plain
    | "#" ~"{"                  -- hash

  sstringText
    = ~("\\" | "'") any

  escape
    = "\""  -- quote
    | "\\"  -- slash
    | "n"   -- newline
    | "t"   -- tab

  sescape
    = "'"   -- quote
    | "\\"  -- slash
    | "n"   -- newline
    | "t"   -- tab

  boolean
    = "true"   -- true
    | "false"  -- false

  self
    = "self" ~alnum

  ivar
    = "@" ident

  keyword
    = ("class"
      | "def"
      | "do"
      | "else"
      | "end"
      | "false"
      | "if"
      | "let"
      | "self"
      | "then"
      | "true") ~alnum

  ident  (an identifier)
    = ~keyword letter alnum*

  name
    = letter alnum*

  number  (a number)
    = digit* "." digit+  -- fract
    | digit+             -- whole

  comment
    = "#" (~"\n" any)*

  space += comment
}
