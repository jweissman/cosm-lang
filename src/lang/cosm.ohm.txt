Cosm {
  Program
    = StatementList

  StatementList
    = Statement StatementTail* ";"?

  StatementTail
    = ";" Statement

  Statement
    = ClassStmt
    | DefStmt
    | LetStmt
    | AssignStmt
    | RequireStmt
    | BareCallStmt
    | Exp

  BareCallStmt
    = BareCallee BareCallArgs TrailingBlock?  -- args
    | BareCallee TrailingBlock                -- block

  BareCallee
    = ident BareCalleeTail*

  BareCalleeTail
    = "." name

  ClassStmt
    = kw_class const_name ClassSuper? DoOpt ClassBody kw_end

  ClassSuper
    = "<" const_name

  ClassBody
    = ClassMember*

  ClassMember
    = ClassMetaStmt ";"?  -- meta
    | ClassDefStmt ";"?   -- def

  ClassMetaStmt
    = kw_class "<<" kw_self DoOpt ClassMetaBody kw_end

  ClassMetaBody
    = ClassMetaMember*

  ClassMetaMember
    = ClassMetaDefStmt ";"?

  ClassMetaDefStmt
    = kw_def ident "(" ParamList? ")" "=" Exp                       -- inline_params
    | kw_def ident "=" Exp                                          -- inline
    | kw_def ident "(" ParamList? ")" DoOpt StatementList kw_end   -- block

  ClassDefStmt
    = kw_def kw_self "." ident "(" ParamList? ")" "=" Exp                     -- class_inline_params
    | kw_def kw_self "." ident "=" Exp                                        -- class_inline
    | kw_def kw_self "." ident "(" ParamList? ")" DoOpt StatementList kw_end  -- class
    | kw_def ident "(" ParamList? ")" "=" Exp                                 -- instance_inline_params
    | kw_def ident "=" Exp                                                    -- instance_inline
    | kw_def ident "(" ParamList? ")" DoOpt StatementList kw_end              -- instance

  DefStmt
    = kw_def ident "(" ParamList? ")" "=" Exp   -- inline_params
    | kw_def ident "=" Exp                      -- inline
    | kw_def ident "(" ParamList? ")" DoOpt StatementList kw_end  -- block

  LetStmt
    = kw_let ident "=" Exp

  AssignStmt
    = ident "=" Exp

  RequireStmt
    = kw_require string          -- bare
    | kw_require "(" string ")"  -- paren

  DoOpt
    = kw_do?

  Exp
    = IfExp

  IfExp
    = kw_if Exp kw_then StatementList kw_else StatementList kw_end  -- full
    | CondExp

  CondExp
    = OrExp "?" Exp ":" Exp  -- ternary
    | OrExp

  OrExp
    = OrExp "||" AndExp   -- or
    | AndExp

  AndExp
    = AndExp "&&" CmpExp  -- and
    | CmpExp

  CmpExp
    = AddExp "~=" AddExp  -- semanticEq
    | AddExp "==" AddExp  -- eq
    | AddExp "!=" AddExp  -- neq
    | AddExp "<=" AddExp  -- lte
    | AddExp "<" AddExp   -- lt
    | AddExp ">=" AddExp  -- gte
    | AddExp ">" AddExp   -- gt
    | AddExp

  AddExp
    = AddExp "+" MulExp  -- plus
    | AddExp "-" MulExp  -- minus
    | MulExp

  MulExp
    = MulExp "*" ExpExp  -- times
    | MulExp "/" ExpExp  -- divide
    | ExpExp

  ExpExp
    = UnaryExp "^" ExpExp  -- power
    | UnaryExp

  UnaryExp
    = "!" UnaryExp   -- not
    | "+" UnaryExp   -- pos
    | "-" UnaryExp   -- neg
    | PostExp

  PostExp
    = PostExp "::" const_name  -- const_access
    | PostExp "." name  -- access
    | PostExp "(" CallArgs? ")" TrailingBlock?  -- call
    | PriExp

  PriExp
    = "(" Exp ")"  -- paren
    | kw_require string          -- require_bare
    | kw_require "(" string ")"  -- require
    | kw_super "(" CallArgs? ")"  -- super
    | kw_yield "(" CallArgs? ")"  -- yield
    | kw_do StatementList kw_end  -- block
    | "->" "(" ParamList? ")" "{" StatementList "}"  -- lambda
    | "[" ArrayItems? "]"  -- array
    | "{" HashItems? "}"  -- hash
    | hstring
    | string
    | symbol
    | boolean
    | self
    | ivar
    | ident
    | number

  symbol
    = ":" name

  CallArgs
    = Exp ("," Exp)*

  TrailingBlock
    = kw_do BlockParams? StatementList kw_end

  BlockParams
    = "|" ParamList? "|"

  BareCallArgs
    = BareArg ("," BareArg)*

  BareArg
    = string
    | symbol
    | boolean
    | self
    | ivar
    | ident
    | number
    | BareArray
    | BareHash

  BareArray
    = "[" ArrayItems? "]"

  BareHash
    = "{" HashItems? "}"

  ParamList
    = Param ("," Param)*

  Param
    = "*" ident      -- rest
    | ident "=" Exp  -- default
    | ident          -- plain

  ArrayItems
    = Exp ("," Exp)*

  HashItems
    = HashEntry ("," HashEntry)*

  HashEntry
    = ident ":" Exp  -- pair
    | ident          -- shorthand

  string
    = dstring
    | sstring

  hstring
    = "\"\"\"" hstringPart* "\"\"\""

  dstring
    = "\"" stringPart* "\""

  sstring
    = "'" sstringPart* "'"

  stringPart
    = interpolation  -- interp
    | stringText     -- text
    | "\\" escape    -- escape

  sstringPart
    = sstringText   -- text
    | "\\" sescape  -- escape

  hstringPart
    = interpolation  -- interp
    | hstringText    -- text
    | "\\" escape    -- escape

  interpolation
    = "#{" applySyntactic<Exp> "}"

  stringText
    = ~("\\" | "\"" | "#") any  -- plain
    | "#" ~"{"                  -- hash

  sstringText
    = ~("\\" | "'") any

  hstringText
    = ~("\\" | "\"\"\"" | "#") any  -- plain
    | "#" ~"{"                      -- hash

  escape
    = "\""  -- quote
    | "\\"  -- slash
    | "n"   -- newline
    | "t"   -- tab

  sescape
    = "'"   -- quote
    | "\\"  -- slash
    | "n"   -- newline
    | "t"   -- tab

  boolean
    = kw_true   -- true
    | kw_false  -- false

  self
    = kw_self

  kw_class = "class" ~alnum
  kw_def = "def" ~alnum
  kw_do = "do" ~alnum
  kw_else = "else" ~alnum
  kw_end = "end" ~alnum
  kw_false = "false" ~alnum
  kw_if = "if" ~alnum
  kw_let = "let" ~alnum
  kw_require = "require" ~alnum
  kw_self = "self" ~alnum
  kw_super = "super" ~alnum
  kw_then = "then" ~alnum
  kw_true = "true" ~alnum
  kw_yield = "yield" ~alnum

  ivar
    = "@" ident

  keyword
    = kw_class
    | kw_def
    | kw_do
    | kw_else
    | kw_end
    | kw_false
    | kw_if
    | kw_let
    | kw_require
    | kw_self
    | kw_super
    | kw_then
    | kw_true
    | kw_yield

  ident  (an identifier)
    = ~keyword identStart identRest*

  name
    = identStart identRest*

  const_name
    = upper identRest*

  identStart
    = letter
    | "_"

  identRest
    = alnum
    | "_"

  number  (a number)
    = digit* "." digit+  -- fract
    | digit+             -- whole

  comment
    = "#" (~"\n" any)*

  space += comment
}
